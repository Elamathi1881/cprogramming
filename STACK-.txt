#include <stdio.h>
#define size  5
int array[size];
int top  = -1;
//push
void push(int value){
    if(top == size -1){
        printf("No space left \n");
    }
    else{
        top++;
        array[top] = value;
        printf("The value %d pushed to array \n", value);
    }
}

//pop
void pop(){
    if(top == -1){
        printf("No value at stack");
    }
    else{
        printf("The top value is: %d \n", array[top]);
        top--;
    }
}
void peak(){
    if(top == -1){
        printf("No value at stack");
    }
    else{
        printf("The peak value is: %d \n", array[top]);
        
    }
}

void isempty(){
    if(top == -1){
        printf("No value at stack \n");
    }
    else{
        printf("stack has items \n");
        
    }
}
void Size(){
    printf("My stack size is %d \n", top +1);
}
int main() {
    push(10);
isempty();
Size();
pop();
Size();
    return 0;
}

#include <stdio.h>

#define MAX 10

int stack[MAX];
int top = -1;

/* push action */
void push(int action) {
    if (top == MAX - 1) {
        printf("Stack Overflow! Cannot save action\n");
    }
    top++;
    stack[top] = action;
    printf("Action %d saved\n", stack[top]);
}

/* undo action */
void pop() {
    if (top == -1) {
        printf("No action to undo\n");
    }
    top--;
    printf("Undo action %d\n", stack[top]);
}

/* last action */
void peek() {
    if (top == -1) {
        printf("No recent action\n");
        return;
    }
    printf("Last action: %d\n", stack[top]);
}

/* check empty */
void isEmpty() {
    if (top == -1)
        printf("Undo not possible (stack empty)\n");
    else
        printf("Undo possible\n");
}
void size() {
    printf("Actions available to undo: %d\n", top + 1);
}

int main() {
    push(1);  
    push(2);   
    push(3);   

    pop();     
    peek();    
    size();    
    isEmpty();

    return 0;
}
