ENQUEUE:-

#include<stdio.h>
#define size 5
int array[size];
int top=-1;
int bottom=-1;
void enqueue(int value){
    if(bottom==size-1){
        printf(" no space in queue\n");
    }else{
        if(top==-1){
            top=0;
        }
        bottom++;
        array[bottom]=value;
        printf("the value is %d added in queue\n",array[bottom]);
    }
}
int main(){
    enqueue(10);
    enqueue(20);
    return 0;
}

DEQUEUE:-

void dequeue(){
    if(top==-1||top>bottom){
        printf("No data in queue\n");
        
    }else{
        printf("The value %d is removed from queue",array[top]);
        top++;
    }
}
int main(){
    enqueue(10);
    enqueue(20);
    dequeue();
    return 0;
}
