single linked list:-


#include <stdio.h>
#include<stdlib.h>
struct contact{
    long int number;
    struct contact*next;
};

int main() {
    struct contact*st1=(struct contact*)malloc(sizeof(struct contact));
    struct contact*st2=(struct contact*)malloc(sizeof(struct contact));
    struct contact*st3=(struct contact*)malloc(sizeof(struct contact));
     st1->number=1234567894;
     st2->number=1254789630;
     st3->number=7894561230;
     st1->next=st2;
     st2->next=st3;
     st3->next=NULL;
     struct contact*head=st1;
     printf("contact\n");
     while(head!=NULL){
         printf("%ld \n",head->number);
         head=head->next;
     }
     printf("finish\n");
    return 0;
}
