 #include <stdio.h>
#include <stdlib.h>

struct Node {
    int songID;
    struct Node *prev;
    struct Node *next;
};

struct Node *head = NULL;

// Insert at beginning
void insertBegin(int value) {
    struct Node *newNode = (struct Node *)malloc(sizeof(struct Node));

    newNode->songID = value;
    newNode->prev = NULL;
    newNode->next = head;

    if (head != NULL) {
        head->prev = newNode;
    }

    head = newNode;
}

// Display playlist forward
void display() {
    struct Node *temp = head;
    while (temp != NULL) {
        printf("%d <-> ", temp->songID);
        temp = temp->next;
    }
    printf("NULL");
}

int main() {
    insertBegin(30);
    insertBegin(20);
    insertBegin(10);

    display();
    return 0;
}
